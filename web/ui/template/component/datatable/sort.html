{{define "com.datatable.sort"}} {{$key := .Props.Key}} {{$direction := .Props.Direction}} {{if ne $direction "desc"}} {{$direction = "asc"}} {{end}} {{if not $key}}{{.Log.Warn "com.datatable.sort: missing required prop: Key"}}{{end}} {{$value := printf "%v.%v" $key $direction}} {{$arrow := "▼"}} {{$sort := .URL.Query.Get "sort"}} {{$isActive := HasPrefix $sort (printf "%v." $key)}} {{if $isActive}} {{if HasSuffix $sort ".asc"}} {{if ne $direction "asc"}} {{$value = ""}} {{else}} {{$value = printf "%v.desc" $key}} {{end}} {{$arrow = "▼"}} {{else}} {{if ne $direction "desc"}} {{$value = ""}} {{else}} {{$value = printf "%v.asc" $key}} {{end}} {{$arrow = "▲"}} {{end}} {{end}} {{$class := "data-table__sort"}} {{if $isActive}} {{$class = printf "%v data-table__sort--active" $class}} {{end}} {{$arrow}} {{end}}
